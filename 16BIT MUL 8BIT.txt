16BIT x 8BIT = 24BIT
INPUT: OPD1_H:OPD1_L : SO BI NHAN 16BIT
	 OPD2 : SO NHAN 8BIT
OUPUT: OPD2:OPD1_H:0PD1_L
	  8     8     8
DUNG R2,R3
;------------------------------------

MUL16_8: MUL OPD1_L,OPD2
MOVW R2,R0
MUL OPD1_H,OPD2
ADD R3,R0
MOV OPD1_H,R3 ;[15:8]
CLR OPD2
ADC OPD2,R1 ;[23:16]
MOV OPD1_L,R2 ;[7:0]
RET

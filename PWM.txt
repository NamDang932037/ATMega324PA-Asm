SBI DDRB,4 ;OC0B OUTPUT
MAIN:   CALL DELAY_10MS	
		LDI R16,124
		OUT OCR0A,R16 ;TAO SONG 1KHZ
		LDI R20,PMIN
		OUT OCR0B,R20 ;GIA TRI DC% MIN

		LDI R16,0B10100011 ;TIMER0 MODE FAST7
		OUT TCCR0A,R16
		LDI R16,0B00001011;TIMER0 HE SO CHIA 64
		OUT TCCR0B,R16

LOOP_INC: 
		RCALL DELAY_10MS
		INC R20
		OUT OCR0B,R20
		CPI R20,125
		BRNE LOOP_INC

LOOP_DEC:
		RCALL DELAY_10MS
		DEC R20
		OUT OCR0B,R20
		CPI R20,0
		BRNE LOOP_DEC
		
		RJMP MAIN
